    <script>
  var directionsDisplay;
  var directionsService = new google.maps.DirectionsService();
  var map;
function initialize() {
  var startLocation = new google.maps.LatLng(<%= @start_lat %>,<%=@start_lng%>)
  var endLocation = new google.maps.LatLng(<%=@end_lat%>,<%=@end_lng%>)
  var mapOptions = {
    zoom: 14,
    center: new google.maps.LatLng(40.7031,-73.9334)
  }
  var map = new google.maps.Map(document.getElementById('map-canvas'), mapOptions);


  new google.maps.Marker({
    position: startLocation,
    map: map,
    icon: 'https://cdn2.iconfinder.com/data/icons/free-basic-icon-set-2/300/2-32.png',
    title: "Start"
  });

  new google.maps.Marker({
    position: endLocation,
    map: map,
    title: "End"
  });

  <% @start_locations.each do |location| %>
    new google.maps.Marker({
      position: new google.maps.LatLng(<%= location[:lat] %>,<%= location[:long] %>),
      map: map,
      icon: 'https://cdn1.iconfinder.com/data/icons/Momentum_MatteEntireSet/32/bullet-blue.png',
      title: "<%= location[:name] %>"
    });
  <% end %>

  <% @mid_locations.each do |location| %>
    new google.maps.Marker({
      position: new google.maps.LatLng(<%= location[:lat] %>,<%= location[:long] %>),
      map: map,
      icon: 'https://cdn1.iconfinder.com/data/icons/Momentum_MatteEntireSet/32/bullet-blue.png',
      title: "<%= location[:name] %>"
    });
  <% end %>

  <% @end_locations.each do |location| %>
    new google.maps.Marker({
      position: new google.maps.LatLng(<%= location[:lat] %>,<%= location[:long] %>),
      map: map,
      icon: 'https://cdn1.iconfinder.com/data/icons/Momentum_MatteEntireSet/32/bullet-blue.png',
      title: "<%= location[:name] %>"
    });
  <% end %>
  directionsDisplay = new google.maps.DirectionsRenderer();
  directionsDisplay.setMap(map);
}

function calcRoute() {
  var start = "297 Devoe Street, Brooklyn, NY"
  var end = "946 Bushwick Ave, Brooklyn, NY"

  var request = {
      origin:start,
      destination:end,
      travelMode: google.maps.TravelMode.DRIVING
  };
  directionsService.route(request, function(response, status) {
    if (status == google.maps.DirectionsStatus.OK) {
      directionsDisplay.setDirections(response);
    }
  });

    google.maps.event.addDomListener(window, 'load', initialize);
    google.maps.event.addDomListener(window, 'load', calcRoute);
